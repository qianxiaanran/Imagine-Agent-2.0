$(function() {
    var ua = navigator.userAgent.toLowerCase();
    if (/windows/.test(ua)) $('#windows_download').show();
    // locate
    $('#locate').on('click', getPosition);

    $(window).resize(setSkin);
    setSkin();

    $(document).on('click', function(e) {
        if( $(e.target).attr('id') == 'select_year' ) {
            $('.select_months').hide();
        } else if ( $(e.target).attr('id') == 'select_month' ) {
            $('.calendar_select_years').hide();
        } else {

        }
    });
    var cityWidth = '';
    // select city
    $('.search_default b').on('click', function() {
        $('.search_default em').width() ? cityWidth = $('.search_default em').width() : '';

            'paddingLeft': '0px',
            'width': '0px',
        }, 500);
        $('#search .search_default input').animate({
            'width': '370px',
            'opacity': '1',
        }, 500);
        $('#search .search_default i').animate({
            'width': '20px',
            'marginRight': '10px',
            'opacity': '1',
        });
        $('#search .search_default input').focus();
    });
    $('.search_default input').on('keyup', function() {
        $('.search_city').show();
        getCitys($(this).val());
    });
    $('#search .search_default i').on('click', function() {
        $(this).animate({
            'width': '0px',
            'marginRight': '0px',
        }, 500);
        $('.search_default em').animate({
            'paddingLeft': '30px',
            'width': cityWidth,
        }, 500);
        $('#search .search_default input').animate({
            'width': '0',
            'opacity': '0',
        }, 500);
        $('#search .search_default i').animate({
            'opacity': '0',
        });
        $('#search .search_default input').val('');
        $('.search_city').hide();
    });
    $('#search .search_city li').on('click', function() {
        $('.search_input').hide();
        $('.search_default').show().text($(this).text());
        $('#search .search_input input').val('');
        $('.search_city').hide();
    });

    var date = new Date( ( new Date() ).setDate( (new Date() ).getDate() + 15 ) );
    // pop year select
    $('.select_year').on('click', function() {
        var year = date.getFullYear();
        var yearArr = new Array();
        var start = 2015;
        for(var i = 0; i <= year - 2015; i++) {
            var li = '<li data="' + start +'">' + start + '年</li>';
            yearArr.push(li);
            start ++;
        }
        $('.select_years').html(yearArr).show();
    });
    // select year
    $('.select_year').on('click', '.select_years li', function() {
        $('#select_year').text($(this).text());
        $('.select_years').hide();
        getCalendar();
    });
    // pop month select
    $('#select_month').on('click', function() {
        $('.select_months').show();
    });
    // select month
    $('.select_month').on('click', '.select_months li', function() {
        $('.select_months').hide();
        $('#select_month').text($(this).text());
        getCalendar();
    });
    // return today
    $('.return_today').on('click', function() {
        var today = new Date();
        var tY = today.getFullYear();
        var tM = today.getMonth() * 1 + 1;
        var tM = tM > 9 ? tM : '0' + tM;
        $('.select_year em').text(tY + '年');
        $('.select_month em').text(tM + '月');
        getCalendar();
    });

    // set liveview picture
    setLiveviewPic();
});

function getCalendar() {
    var y = $('.select_year em').text();
    var m = $('.select_month em').text();
    var y = y.match(/\d{4}/);
    var m = m.match(/\d{2}/);
    var ym = y + '-' + m;
    $.ajax({
        url: '/api/month/' + ym,
        async: false,
        success: function(data) {
            var data = JSON.parse(data);
            if(data.code != 1) {
                $('#calendar_grid').html(data.data.html);
                $('#calendar_grid .active').addClass('star'); 
            } else {
                $('#calendar_grid').html('<div class="calendar_tips">没有数据！</div>');
            }
            $('#month_lunardate').html(data.data.month_lunardate);
        },
        error: function(data) {
            console.log(data.responseText);
       }
    });
}

function getCitys(city) {
    $.ajax({
        url: '/api/citysearch/' + city,
        async: true,
        success: function(data) {
            var data = JSON.parse(data);
            if(data.rc.c == '0' && data.city_list != undefined) {
                setSearchCitys(data.city_list);
            }
        },
        error: function(data) {
            console.log(data);
       }
    });
}
function setSearchCitys(citys) {
    var arrCitys = new Array ();
    for(var i = 0; i < citys.length; i++) {
        if (citys[i].counname == '中国') {
            var url = '/weather/' + citys[i].counname + '/' + citys[i].pname + '/' +citys[i].name;
            var url = url.toLowerCase();
            var url = url.replace(/\s/g, '_');
            var li = '<li><a href="/api/redirect/' +citys[i].cityId + '">' +
                    citys[i].name + ',' +citys[i].pname + ',' + citys[i].counname + '</a></li>';
            arrCitys.push(li);
        }
    }
    $('.search_city ul').html(arrCitys);
}
function setSkin() {
    var skinEl = $('#skin');
    var img = new Image();
    var weburl = window.location.host;
    img.src = skinEl.attr('data-url');
    img.onload = function() {
        var wSkin = skinEl.width();
        var hSkin = skinEl.height();
        var wImg = img.width;
        var hImg = img.height;
        var scale = wSkin / hSkin - wImg / hImg;
        if (scale >= 0) {
            if(wSkin >= wImg) {
                skinEl.css({
                    'background-size': wSkin + 'px auto'
                });
            } else {
                skinEl.css({
                    'background-size': 'auto auto'
                });
            }
        } else {
            if(hSkin >= hImg) {
                skinEl.css({
                    'background-size': 'auto ' + hSkin + 'px'
                });
            } else {
                skinEl.css({
                    'background-size': 'auto auto'
                });
            }
        }
    }
}

var showCity = function(position) {
    var longitude = position.coords.longitude;
    var latitude = position.coords.latitude;

    if (longitude == undefined) return false;
    $.ajax({
        url: '/api/geolocate/'+longitude +'/'+latitude,
        data: {},
        success: function(data) {
            $('#locate').css({'background': 'url(/assets/images/icon/i_locate.png) no-repeat right center'});
            if (data != 0){
                location = data;
            }
        },
        error: function(data) {
            console.log(data);
       }
    });
}
function getPosition() {
    $('#locate').css({'background': 'url(/assets/images/icon/i_loading.png) no-repeat right center'});
    if(navigator.geolocation) {
        // safari ok
        navigator.geolocation.getCurrentPosition(showCity, function(error){
        $.ajax({
            url: '/api/iplocate',
            data: {},
            success: function(data) {
                console.log(data);
                $('#locate').css({'background': 'url(/assets/images/index/i_location.png) no-repeat right center'});
                if (data != 0){
                    location = data;
                } else {
                    alert('定位失败!');
                }
            },
            error: function(data) {
                console.log(data);
            }
        });

        }, {timeout:3000});
    } else {
        console.log('navigator not work');
    }
}
function setLiveviewPic() {
    var lv = $('.liveview .item li a img');
    var wSkin = 224;
    var hSkin = 150;
    for(var i = 0; i < lv.length; i++) {
        var wImg = wSkin;
        var hImg = parseInt( lv.eq(i).attr('data-height') * wSkin / lv.eq(i).attr('data-width') );
        if(hImg >= hSkin) {
            lv.eq(i).css({
                width: wSkin + 'px',
                height: 'auto'
            });
        } else {
            lv.eq(i).css({
                width: wSkin + 'px',
                height: hSkin + 'px',
            });
        }
    }
};
